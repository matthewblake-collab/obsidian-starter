Manual path — save your credentials securely
Set your Telegram tokens as environment variables. Your bot reads them at startup. You do this once and never touch them again.

What is an environment variable? A saved value your Mac remembers permanently. Instead of putting your secret token directly in the bot code (dangerous), the Mac holds it separately and passes it to the bot when it starts. You set it once in a file called .zshrc and it's available forever.

Open .zshrc and add your credentials:

#!/usr/bin/env python3
"""Jarvis — AI Telegram bot. Commands: /status /briefing /help"""
import json, os, subprocess, sys, time, urllib.request, urllib.error
from datetime import datetime, timezone, timedelta

BOT_TOKEN  = os.environ.get("TELEGRAM_BOT_TOKEN","")
CHAT_ID    = int(os.environ.get("TELEGRAM_CHAT_ID","0"))
CLAUDE_BIN = os.environ.get("CLAUDE_BIN", os.path.expanduser("~/.local/bin/claude"))
YOUR_NAME  = os.environ.get("JARVIS_YOUR_NAME","Boss")
YOUR_WORK  = os.environ.get("JARVIS_YOUR_WORK","a founder building a product")
TZ_OFFSET  = int(os.environ.get("JARVIS_TZ_OFFSET","10"))
MY_TZ      = timezone(timedelta(hours=TZ_OFFSET))
STATE_FILE = os.path.expanduser("~/jarvis/state.json")
WORKDIR    = os.path.expanduser("~")

SYSTEM_CONTEXT = f"""You are Jarvis, AI assistant for {YOUR_NAME}.
They work on: {YOUR_WORK}.
Rules: call them {YOUR_NAME}. Short replies (phone). Calm, capable, no fluff."""

FEATURES_PROMPT = f"""Search web for new Anthropic features in the last 7 days:
- https://docs.anthropic.com/en/release-notes/overview
- https://www.anthropic.com/news
Rate each: HIGH/MEDIUM/LOW for relevance to: {YOUR_WORK}
Format: Item: description — Relevance: LEVEL — reason
If nothing new: No new Anthropic features since last check."""

MORNING_PROMPT = f"""Give {YOUR_NAME} a morning briefing. Emoji bullets, max 2 lines each.
1. ANTHROPIC: New features this week? Rate HIGH/MEDIUM/LOW for: {YOUR_WORK}
2. TODAY: What day and anything notable?
End with: Reply /help for commands."""

def log(m): print(f"[{datetime.now().isoformat(timespec='seconds')}] {m}",flush=True)
def tg(method, **p):
    url=f"https://api.telegram.org/bot{BOT_TOKEN}/{method}"
    req=urllib.request.Request(url,json.dumps(p).encode(),{"Content-Type":"application/json"})
    with urllib.request.urlopen(req,timeout=30) as r: return json.loads(r.read())
def send(text):
    if len(text)>4000: text=text[:3997]+"..."
    try: tg("sendMessage",chat_id=CHAT_ID,text=text)
    except Exception as e: log(f"send failed: {e}")
def run_claude(prompt, tools="Read,WebSearch,WebFetch", timeout=120):
    r=subprocess.run([CLAUDE_BIN,"--print",prompt,"--allowedTools",tools],
        capture_output=True,text=True,timeout=timeout,cwd=WORKDIR)
    if r.returncode!=0: raise RuntimeError(r.stderr[:200])
    return r.stdout.strip()
def ask_claude(text, history):
    convo="".join(f"{'User' if m['role']=='user' else 'Jarvis'}: {m['content']}\n" for m in history[-10:])
    return run_claude(f"{SYSTEM_CONTEXT}\n\nConversation:\n{convo}User: {text}\nJarvis:")
def cmd_help():
    return f"Jarvis Commands\n\n/briefing — morning briefing\n/status   — system status\n/help     — this list\n\nOr ask me anything."
def cmd_briefing():
    send("Running briefing..."); return run_claude(MORNING_PROMPT)
def cmd_status():
    lines=[f"Status — {datetime.now(tz=MY_TZ).strftime('%a %d %b %H:%M')}"]
    r=subprocess.run([CLAUDE_BIN,"--version"],capture_output=True,text=True)
    lines.append(f"Claude: {'OK' if r.returncode==0 else 'ERROR'}")
    try:
        s=json.loads(open(STATE_FILE).read())
        lines.append(f"Last briefing: {s.get('last_features_check','never')}")
    except: pass
    return "\n".join(lines)
COMMANDS={"/help":cmd_help,"/start":cmd_help,"/briefing":cmd_briefing,"/morning":cmd_briefing,"/status":cmd_status}
def load_state():
    try: return json.loads(open(STATE_FILE).read())
    except: return {}
def save_state(s):
    with open(STATE_FILE,"w") as f: json.dump(s,f,indent=2)
def should_run(state, key, hour):
    now=datetime.now(tz=MY_TZ)
    return now.hour==hour and state.get(key)!=now.strftime("%Y-%m-%d")
def main():
    if not BOT_TOKEN or not CHAT_ID:
        log("FATAL: TELEGRAM_BOT_TOKEN and TELEGRAM_CHAT_ID required"); sys.exit(1)
    if not os.path.exists(CLAUDE_BIN):
        log(f"FATAL: claude not found at {CLAUDE_BIN}. Run: which claude"); sys.exit(1)
    log(f"Jarvis online — chat {CHAT_ID}, tz UTC{TZ_OFFSET:+d}")
    send(f"Jarvis online, {YOUR_NAME}. Type /help for commands.")
    offset,history,state=0,[],load_state()
    while True:
        try:
            if should_run(state,"last_features_check",8):
                today=datetime.now(tz=MY_TZ).strftime("%Y-%m-%d")
                state["last_features_check"]=today; save_state(state)
                try: send(run_claude(FEATURES_PROMPT))
                except Exception as e: send(f"Daily check failed: {e}")
            resp=tg("getUpdates",offset=offset,timeout=20,allowed_updates=["message"])
            for upd in resp.get("result",[]):
                offset=upd["update_id"]+1
                msg=upd.get("message",{})
                if msg.get("chat",{}).get("id")!=CHAT_ID: continue
                text=msg.get("text","").strip()
                if not text: continue
                cmd_key=text.split()[0].lower() if text.startswith("/") else None
                try:
                    if cmd_key and cmd_key in COMMANDS: reply=COMMANDS[cmd_key]()
                    else:
                        reply=ask_claude(text,history)
                        history.append({"role":"user","content":text})
                        history.append({"role":"assistant","content":reply})
                        history=history[-20:]
                    send(reply)
                except Exception as e: send(f"Error: {e}")
        except urllib.error.URLError as e: log(f"Network: {e}"); time.sleep(5)
        except Exception as e: log(f"Error: {e}"); time.sleep(5)
if __name__=="__main__": main()
PYEOF
